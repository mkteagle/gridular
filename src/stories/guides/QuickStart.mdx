# Quick Start Guide

Get up and running with the Virtualized Grid in 5 minutes.

## Installation

```bash
# Install the package (example - adjust for your setup)
npm install @your-org/virtualized-grid
```

## Basic Usage

### 1. Define Your Columns

```tsx
import { ColumnDef } from "./types";

const columns: ColumnDef[] = [
  {
    id: "id",
    key: "id",
    header: "ID",
    width: 80,
  },
  {
    id: "name",
    key: "name",
    header: "Name",
    width: 200,
  },
  {
    id: "email",
    key: "email",
    header: "Email",
    width: 250,
  },
];
```

### 2. Prepare Your Data

```tsx
const data = [
  { id: 1, name: "John Doe", email: "john@example.com" },
  { id: 2, name: "Jane Smith", email: "jane@example.com" },
  // ... more rows
];
```

### 3. Render the Grid

```tsx
import { VirtualizedGridEnhancedV2 } from "./VirtualizedGridEnhancedV2";

function MyComponent() {
  return (
    <div style={{ height: "600px" }}>
      <VirtualizedGridEnhancedV2 columns={columns} data={data} />
    </div>
  );
}
```

## Adding Features

### Enable Sorting

```tsx
<VirtualizedGridEnhancedV2
  columns={columns}
  data={data}
  enableSorting
  sortState={sortState}
  onSortChange={setSortState}
/>
```

### Enable Pagination

```tsx
const [pageIndex, setPageIndex] = useState(0);
const [pageSize, setPageSize] = useState(20);

<VirtualizedGridEnhancedV2
  columns={columns}
  data={data}
  pagination={{
    pageIndex,
    pageSize,
    totalRows: data.length,
    onPageChange: setPageIndex,
    onPageSizeChange: setPageSize,
  }}
/>;
```

### Enable Grouping

```tsx
const [groupingState, setGroupingState] = useState({
  groupByColumns: [],
  expandedGroups: {},
});

<VirtualizedGridEnhancedV2
  columns={columns}
  data={data}
  enableGrouping
  groupingState={groupingState}
  onGroupingChange={setGroupingState}
/>;
```

### Enable Column Resizing

```tsx
const [columnWidths, setColumnWidths] = useState({});

<VirtualizedGridEnhancedV2
  columns={columns}
  data={data}
  enableColumnResize
  columnWidths={columnWidths}
  onColumnWidthsChange={setColumnWidths}
/>;
```

## Complete Example

```tsx
import { useState, useMemo } from "react";
import { VirtualizedGridEnhancedV2 } from "./VirtualizedGridEnhancedV2";
import type { ColumnDef, SortState } from "./types";

function DataGrid() {
  // State
  const [sortState, setSortState] = useState<SortState | null>(null);
  const [pageIndex, setPageIndex] = useState(0);
  const [pageSize, setPageSize] = useState(20);

  // Columns definition
  const columns: ColumnDef[] = useMemo(
    () => [
      { id: "id", key: "id", header: "ID", width: 80, enableSorting: true },
      {
        id: "name",
        key: "name",
        header: "Name",
        width: 200,
        enableSorting: true,
      },
      { id: "email", key: "email", header: "Email", width: 250 },
      {
        id: "status",
        key: "status",
        header: "Status",
        width: 120,
        enableSorting: true,
        render: (row) => (
          <span
            className={
              row.status === "active" ? "text-green-600" : "text-gray-400"
            }
          >
            {row.status}
          </span>
        ),
      },
    ],
    []
  );

  // Data (use useMemo to prevent regeneration on re-renders!)
  const data = useMemo(
    () => [
      { id: 1, name: "John Doe", email: "john@example.com", status: "active" },
      {
        id: 2,
        name: "Jane Smith",
        email: "jane@example.com",
        status: "inactive",
      },
      // ... more rows
    ],
    []
  );

  return (
    <div style={{ height: "600px" }}>
      <VirtualizedGridEnhancedV2
        columns={columns}
        data={data}
        enableSorting
        sortState={sortState}
        onSortChange={setSortState}
        enableColumnResize
        pagination={{
          pageIndex,
          pageSize,
          totalRows: data.length,
          onPageChange: setPageIndex,
          onPageSizeChange: setPageSize,
        }}
        gridId="my-data-grid"
      />
    </div>
  );
}
```

## Important Notes

### ‚ö†Ô∏è Performance Tip: Use `useMemo` for Data

Always wrap your data generation in `useMemo` to prevent it from being regenerated on every render:

```tsx
// ‚úÖ Good
const data = useMemo(() => generateData(1000), []);

// ‚ùå Bad - regenerates on every render!
const data = generateData(1000);
```

### üìè Container Height Required

The grid requires a fixed height container to enable virtualization:

```tsx
// ‚úÖ Good
<div style={{ height: '600px' }}>
  <VirtualizedGridEnhancedV2 ... />
</div>

// ‚ùå Bad - no height specified
<div>
  <VirtualizedGridEnhancedV2 ... />
</div>
```

### üîë Unique Row IDs

Ensure each row has a unique identifier:

```tsx
// Default: uses 'id' field
<VirtualizedGridEnhancedV2 ... />

// Custom: specify your own ID getter
<VirtualizedGridEnhancedV2
  getRowId={(row) => row.userId}
  ...
/>
```
